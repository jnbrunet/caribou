        PUBLIC "$<BUILD_INTERFACE:${CMAKE_CURRENT_SOURCE_DIR}/src/>"
        PUBLIC $<INSTALL_INTERFACE:include>)
    set_property(TARGET SofaCaribouPython PROPERTY CXX_STANDARD 17)
